## Install

```bash
npm install --save made-by-adam
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'made-by-adam'
import 'made-by-adam/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
